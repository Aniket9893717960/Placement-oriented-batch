class Ts
{
 public static void main(String args[]){
  String str = args[0].toString();
  StringBuilder sb = new StringBuilder();
  for(int i=0;i<str.length();i++){
    char ch = str.charAt(i);
    if(ch == 'a')
    sb.append("*");
    else
    sb.append(ch);
  }  
  System.out.println(sb.toString());
 }
}
